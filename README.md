# exmaple run and usage

```python 
from rubx import rubx

auth : str = ''
robot = rubx.Bot('AppName', auth=auth)

guid : str = ''

def main():
    while True:
        try:
            robot.reportObject(user=guid, mode='spam')
            break
        except:
            pass
            
main()
```
